Setting up Sifr.html

        
          <!DOCTYPE>
          <html>
          <head>
          <META http-equiv="Content-Type" content="text/html; charset=UTF-8">
          </head>
          <body>
          <script type="text/javascript" src="crossdomain-xhr.min.js"></script>
            <script type="text/javascript">
            cXHR.config.set(
            {
            'allowedOrigins' : '*'
          }
          );
          </script>
          </body>
          </html>
